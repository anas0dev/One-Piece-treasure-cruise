001import java.util.ArrayList;
002import java.util.HashMap;
003
004/**
005 * Class Room - a room in an adventure game.
006 *
007 * This class is part of the "World of Zuul" application. 
008 * "World of Zuul" is a very simple, text based adventure game.  
009 *
010 * A "Room" represents one location in the scenery of the game.  It is 
011 * connected to other rooms via exits.  The exits are labelled north, 
012 * east, south, west.  For each direction, the room stores a reference
013 * to the neighboring room, or null if there is no exit in that direction.
014 * 
015 * @author  Michael Kolling and David J. Barnes
016 * @version 2006.03.30
017 */
018class Room
019{
020        private String description;
021        private HashMap<String, Room> exits;
022        private String imageName;
023        private ArrayList<Item> tresors;
024        /**
025        * Create a room described "description". Initially, it
026        * has no exits. "description" is something like "a
027        * kitchen" or "an open court yard".
028        */
029        public Room(String description, String imageName)
030        {
031                this.description = description;
032                exits = new HashMap<String, Room>();
033                this.imageName = imageName;
034                this.tresors = new ArrayList<Item>();
035        }
036        /**
037        * Define the exits of this room. Every direction either
038        * leads to another room or is null (no exit there).
039        */
040        public void setExits(Room north, Room east, Room south,Room west, Room northWest, Room northEast, Room southWest, Room southEast)
041        {
042                if(north != null)
043                        exits.put("north", north);
044                if(east != null)
045                        exits.put("east", east);
046                if(south != null)
047                        exits.put("south", south);
048                if(west != null)
049                        exits.put("west", west);
050                if(northEast != null)
051                        exits.put("northeast", northEast);
052                if(northWest != null)
053                        exits.put("northwest", northWest);
054                if(southEast != null)
055                        exits.put("southeast", southEast);
056                if(southWest != null)
057                        exits.put("southwest", southWest);
058        }
059        
060        public void setExit(String direction, Room voisin)
061        {
062                exits.put(direction, voisin);
063        }
064        /**
065        * Return the room that is reached if we go from this
066        * room in direction "direction". If there is no room in
067        * that direction, return null.
068        */
069        public Room getExit(String direction)
070        {
071                return exits.get(direction);
072        }
073        
074        public String getExitString()
075    {
076        StringBuilder returnString = new StringBuilder("Sorties:");
077        for ( String chaine : exits.keySet() )
078            returnString.append( " " +  chaine);
079        return description + ".\n" + returnString.toString();
080    }
081        public String getLongDescription()
082        {
083                return getExitString();
084        }
085        /**
086        * Return the description of the room (the one that was
087        * defined in the constructor).
088        */
089        public String getDescription()
090        {
091                return description;
092        }
093        /**
094     * Return a string describing the room's image name
095     */
096        public String getImageName()
097        {
098                return imageName;
099        }
100        
101        public void addTresor(Item tresor) 
102        {
103                this.tresors.add(tresor);
104        }
105        
106        public void removeTresor(Item tresor)
107        {
108                this.tresors.remove(tresor);
109        }
110        
111        public String printTresor()
112        {
113                String result = "";
114                for(int i = 0; i < this.tresors.size(); i++) {
115                        result += " " + i + "* " + this.tresors.get(i).getDescription() + "\n";
116                }
117                return result;
118        }
119}
