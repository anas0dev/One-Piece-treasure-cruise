001import javax.swing.*;
002import java.awt.*;
003import java.awt.event.*;
004import java.net.URL;
005//import java.awt.image.*;
006
007/**
008 * This class implements a simple graphical user interface with a text entry
009 * area, a text output area and an optional image.
010 * 
011 * @author Michael Kolling
012 * @version 1.0 (Jan 2003)
013 */
014public class UserInterface implements ActionListener
015{
016    private GameEngine engine;
017    private JFrame myFrame;
018    private JTextField entryField;
019    private JTextArea log;
020    private JLabel image;
021    private JButton ButtonL;
022    private JButton ButtonE;    
023    private JButton ButtonQ;
024    private JButton ButtonH;
025    private JButton help;
026    private JButton RBeamer;
027    private JButton ButtonR;
028    private JButton ButtonA;
029    private JButton ButtonT;
030    private JButton ButtonU;
031
032    /**
033     * Construct a UserInterface. As a parameter, a Game Engine
034     * (an object processing and executing the game commands) is
035     * needed.
036     * 
037     * @param gameEngine  The GameEngine object implementing the game logic.
038     */
039    public UserInterface(GameEngine gameEngine)
040    {
041        engine = gameEngine;
042        createGUI();
043    }
044
045    /**
046     * Print out some text into the text area.
047     */
048    public void print(String text)
049    {
050        log.append(text);
051        log.setCaretPosition(log.getDocument().getLength());
052    }
053
054    /**
055     * Print out some text into the text area, followed by a line break.
056     */
057    public void println(String text)
058    {
059        log.append(text + "\n");
060        log.setCaretPosition(log.getDocument().getLength());
061    }
062
063    /**
064     * Show an image file in the interface.
065     */
066    public void showImage(String imageName)
067    {
068        URL imageURL = this.getClass().getClassLoader().getResource(imageName);
069        if(imageURL == null)
070            System.out.println("image not found");
071        else {
072            ImageIcon icon = new ImageIcon(imageURL);
073            image.setIcon(icon);
074            myFrame.pack();
075        }
076    }
077
078    /**
079     * Enable or disable input in the input field.
080     */
081    public void enable(boolean on)
082    {
083        entryField.setEditable(on);
084        if(!on)
085            entryField.getCaret().setBlinkRate(0);
086    }
087
088    /**
089     * Set up graphical user interface.
090     */
091    private void createGUI()
092    {
093        myFrame = new JFrame("Zork");
094        entryField = new JTextField(70);
095
096        log = new JTextArea();
097        log.setEditable(false);
098        JScrollPane listScroller = new JScrollPane(log);
099        listScroller.setPreferredSize(new Dimension(200, 200));
100        listScroller.setMinimumSize(new Dimension(100,100));
101
102        JPanel panel = new JPanel();
103        JPanel panel1 = new JPanel();
104        JPanel panel2 = new JPanel();
105        JPanel panel3 = new JPanel();
106        JPanel panel4 = new JPanel();
107        JPanel panel5 = new JPanel();
108        image = new JLabel();
109        
110        ButtonL = new JButton("north");
111        ButtonQ = new JButton("east");
112        ButtonH = new JButton("south");
113        ButtonE = new JButton("west");
114        ButtonR = new JButton("north west");
115        ButtonA = new JButton("south west");
116        ButtonT = new JButton("north east");
117        ButtonU = new JButton("south east");
118        help = new JButton("help");
119        RBeamer = new JButton("Quit");
120        
121        panel.setLayout(new BorderLayout());
122        panel.add(image, BorderLayout.NORTH);
123        panel.add(listScroller, BorderLayout.CENTER);
124        panel.add(entryField, BorderLayout.SOUTH);
125        
126        panel4.setLayout(new BorderLayout());
127        panel4.add(ButtonT, BorderLayout.EAST);
128        ButtonT.addActionListener(this);
129        panel4.add(ButtonR, BorderLayout.WEST);
130        ButtonR.addActionListener(this);
131        
132        panel5.setLayout(new BorderLayout());
133        panel5.add(ButtonU, BorderLayout.EAST);
134        ButtonU.addActionListener(this);
135        panel5.add(ButtonA, BorderLayout.WEST);
136        ButtonA.addActionListener(this);
137        
138        panel3.setLayout(new BorderLayout());
139        panel3.add(panel4, BorderLayout.NORTH);
140        panel3.add(panel5, BorderLayout.SOUTH);
141        
142        panel1.setLayout(new BorderLayout());
143        panel1.add(ButtonL, BorderLayout.NORTH);
144        ButtonL.addActionListener(this);
145        panel1.add(panel3, BorderLayout.CENTER);
146        panel1.add(ButtonQ, BorderLayout.WEST);
147        ButtonQ.addActionListener(this);
148        panel1.add(ButtonH, BorderLayout.SOUTH);
149        ButtonH.addActionListener(this);
150        panel1.add(ButtonE, BorderLayout.EAST);
151        ButtonE.addActionListener(this);
152        
153        panel2.setLayout(new BorderLayout(20,50));
154      ///  panel2.add(Horloge, BorderLayout.WEST);
155        panel2.add(help, BorderLayout.EAST);
156        help.addActionListener(this);
157        panel2.add(RBeamer, BorderLayout.SOUTH);
158        RBeamer.addActionListener(this);
159        panel2.add(panel1, BorderLayout.CENTER);
160        panel.add(panel2, BorderLayout.EAST);
161        
162        myFrame.getContentPane().add(panel, BorderLayout.CENTER);
163
164        // add some event listeners to some components
165        myFrame.addWindowListener(new WindowAdapter() {
166            public void windowClosing(WindowEvent e) {System.exit(0);}
167        });
168
169        entryField.addActionListener(this);
170
171        myFrame.pack();
172        myFrame.setVisible(true);
173        entryField.requestFocus();
174    }
175
176    /**
177     * Actionlistener interface for entry textfield.
178     */
179    public void actionPerformed(ActionEvent e) 
180    {
181        if(e.getActionCommand().equals("west"))engine.interpretCommand("go west");
182        else if(e.getActionCommand().equals("south"))engine.interpretCommand("go south");
183        else if(e.getActionCommand().equals("east"))engine.interpretCommand("go east");
184        else if(e.getActionCommand().equals("north"))engine.interpretCommand("go north");
185        else if(e.getActionCommand().equals("north east"))engine.interpretCommand("go northeast");
186        else if(e.getActionCommand().equals("north west"))engine.interpretCommand("go northwest");
187        else if(e.getActionCommand().equals("south east"))engine.interpretCommand("go southeast");
188        else if(e.getActionCommand().equals("south west"))engine.interpretCommand("go southwest");
189        else if(e.getActionCommand().equals("help"))engine.interpretCommand("help");
190        else if(e.getActionCommand().equals("Quit"))engine.interpretCommand("quit");
191        else
192        processCommand();
193    }
194
195    /**
196     * A command has been entered. Read the command and do whatever is 
197     * necessary to process it.
198     */
199    private void processCommand()
200    {
201//        boolean finished = false;
202        String input = entryField.getText();
203        entryField.setText("");
204
205        engine.interpretCommand(input);
206    }
207}
