002import java.util.StringTokenizer;

004/**
005 * This class is part of the "World of Zuul" application. 
006 * "World of Zuul" is a very simple, text based adventure game.  
007 * 
008 * This parser reads user input and tries to interpret it as an "Adventure"
009 * command. Every time it is called it reads a line from the terminal and
010 * tries to interpret the line as a two word command. It returns the command
011 * as an object of class Command.
012 *
013 * The parser has a set of known command words. It checks user input against
014 * the known commands, and if the input is not one of the known commands, it
015 * returns a command object that is marked as an unknown command.
016 * 
017 * @author  Michael Kolling and David J. Barnes
018 * @version 2006.03.30
019 */
020public class Parser 
021{
022    private CommandWords commands;  // holds all valid command words
023 //   private Scanner reader;         // source of command input
024
025    /**
026     * Create a parser to read from the terminal window.
027     */
028    public Parser() 
029    {
030        commands = new CommandWords();
031 //       reader = new Scanner(System.in);
032    }
033    public String showCommands()
034    {
035        return commands.getCommandList();
036    }
037    /**
038     * Get a new command from the user. The command is read by
039     * parsing the 'inputLine'.
040     */
041    public Command getCommand(String inputLine) 
042    {
043        //String inputLine = "";   // will hold the full input line
044        String word1;
045        String word2;
046
047        StringTokenizer tokenizer = new StringTokenizer(inputLine);
048
049        if(tokenizer.hasMoreTokens())
050            word1 = tokenizer.nextToken();      // get first word
051        else
052            word1 = null;
053        if(tokenizer.hasMoreTokens())
054            word2 = tokenizer.nextToken();      // get second word
055        else
056            word2 = null;
057
058        // note: we just ignore the rest of the input line.
059
060        // Now check whether this word is known. If so, create a command
061        // with it. If not, create a "null" command (for unknown command).
062
063        if(commands.isCommand(word1))
064            return new Command(word1, word2);
065        else
066            return new Command(null, word2);
067    }
068}
