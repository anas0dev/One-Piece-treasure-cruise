102    /**
103     * Given a command, process (that is: execute) the command.
104     * If this command ends the game, true is returned, otherwise false is
105     * returned.
106     */
107    public void interpretCommand(String commandLine) 
108    {
109        gui.println(commandLine);
110        Command command = parser.getCommand(commandLine);
111
112        if(command.isUnknown()) {
113            gui.println("Je ne comprend pas ce que vous voulez dire...");
114            return;
115        }
116
117        String commandWord = command.getCommandWord();
118        if (commandWord.equals("help"))
119            printHelp();
120        else if (commandWord.equals("go"))
121            goRoom(command);
122        else if (commandWord.equals("look"))
123            look();
124        else if (commandWord.equals("eat"))
125            eat();
126        else if (commandWord.equals("quit")) {
127            if(command.hasSecondWord())
128                gui.println("Quit what?");
129            else
130                endGame();
131        }
132    }
133    private void printHelp() 
134    {
135        gui.println("Vous êtes perdus sur cette vague merr");
136        gui.print("Vos mots de commande sont " + parser.showCommands());
137    }

139    /** 
140     * Try to go to one direction. If there is an exit, enter the new
141     * room, otherwise print an error message.
142     */
143    private void goRoom(Command command) 
144    {
145        if(!command.hasSecondWord()) {
146            // if there is no second word, we don't know where to go...
147            gui.println("Go where?");
148            return;
149        }
150
151        String direction = command.getSecondWord();
152
153        // Try to leave current room.
154        Room nextRoom = this.currentRoom.getExit(direction);
155
156        if (nextRoom == null)
157            gui.println("There is no door!");
158        else {
159                //this.deplacement.add(new String(direction));
160            this.currentRoom = nextRoom;
161            gui.println(this.currentRoom.getLongDescription());
162            gui.println("Vous remarquez des objet a terre, oh il y a :");
163            gui.println(this.currentRoom.printTresor());
164            if(this.currentRoom.getImageName() != null)
165                gui.showImage(this.currentRoom.getImageName());
166        }
167    }
